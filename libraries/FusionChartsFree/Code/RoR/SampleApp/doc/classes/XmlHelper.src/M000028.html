  def XmlHelper.escape_single_quotes(str_to_escape)
    # Hash containing the required conversion
    conversions = {
       %r{'}=>'%26apos;'
    }  
    escaped_str = str_to_escape
    conversions.each do |x,y|
      escaped_str = escaped_str.gsub(x,y)
    end  
    return escaped_str
  end
